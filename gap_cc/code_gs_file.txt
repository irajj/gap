function userClicked(userInfo){
  var url = "https://docs.google.com/spreadsheets/d/1wg9sFF3RJHxzIH5fVyRzRS836mvVbtS2e-7uerSIFoQ/edit#gid=0";
  var ss = SpreadsheetApp.openByUrl(url);
  var ws = ss.getSheetByName("Data");

  ws.appendRow([userInfo.firstName,userInfo.lastName,userInfo.app, new Date()]);  

  //Logger.log( name + " clicked the btn");
}
